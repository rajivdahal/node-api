database connection and mongoose
1. install mongodb and mongoose
2. make the connection to database
  approach: you have to open connection to the database when the application runs for the first time and leave it don't close the connection 
  so, 
  in app.js when it loads you need to start connection
  at top in app.js:
   require('./db_init')---> coz code will run in app.js at first and connection will establish on another file (seperation of concern)

  redirects to ./db_init
                                const mongodb=require('mongodb)
                                const mongoose=require('mongoose)
                                const dbName="group37db"
                                const conxnURL="mongodb://localhost:27017"+'/'+dbName

                                mongoose.connect(conxnURL,{
                                    useUnifiedTopology: true,
                                    useNewUrlParser: true
                                },
                                function(err,done){
                                    if(err){return console.log("error in db connectionn)} //no next() coz next will only work when url in browser hits 
                                    // but here no url hit so no next but show in console to client that db connection has failed
                                }
                                console.log("db connection successful)
                                )
